# Download the images
for i in tqdm(range(len(compounds_df))):
    row = f'r{compounds_df.loc[i, "row"]}'
    column = f'c{compounds_df.loc[i, "column"]}'
    compound = f'{compounds_df.loc[i, "broad_sample"]}'
    concentration = f'{compounds_df.loc[i, "mmoles_per_liter"]}'

    if not os.path.isdir(f'./prism_compounds/{compound}'):
        cmd = f"./prism_compounds/{compound}_{concentration}"
        os.makedirs(cmd)

    plate = random.choice(plates) # choose a random replicate
    cmd = (
        f"{s3}"
        f"{prefix}"
        f" {bucket}/"
        f"{project}/"
        f"{source}/"
        f"images/"
        f"{batch}/"
        f"images/"
        f"{plate}/"
        f"Images/"
        f" prism_compounds/{compound}_{concentration}/ "
        f'--exclude "*" '
        f'--include "{row}{column}f{field}*.tiff"')
        # f'--dryrun'
    os.system(cmd)
